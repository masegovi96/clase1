Tablas Hash

¿Qué son las tablas hash?

Una tabla hash es una estructura de datos que utiliza una función hash para mapear claves a valores. Es eficiente para operaciones como búsqueda, inserción y eliminación.

Operaciones Básicas

  • put(clave, valor) → inserta un valor asociado a una clave
  • get(clave) → busca el valor
  • remove(clave) → elimina el valor
  • containsKey(clave) → verifica si existe la clave
  • size() → retorna el tamaño de la tabla hash

Implementaciones Comunes

  • Bases de datos
  • Contadores de frecuencia
  • Caches (memoria rápida)

Ejemplo en PHP

class TablaHash {
    private $tabla = [];

    public function insertar($clave, $valor) {
        $indice = $this->hash($clave);
        $this->tabla[$indice] = $valor;
    }

    public function obtener($clave) {
        $indice = $this->hash($clave);
        return $this->tabla[$indice] ?? null;
    }

    private function hash($clave) {
        return crc32($clave) % 10; // Función hash simple
    }
}

// Ejemplo de uso
$tabla = new TablaHash();
$tabla->insertar("clave1", "valor1");
$tabla->insertar("clave2", "valor2");
echo $tabla->obtener("clave1"); // Imprime "valor1"

// Clase TablaHash
class TablaHash {
    constructor() {
        this.tabla = new Array(10);
    }

    hash(clave) {
        let hash = 0;
        for (let i = 0; i < clave.length; i++) {
            hash += clave.charCodeAt(i);
        }
        return hash % 10;
    }

    insertar(clave, valor) {
        const indice = this.hash(clave);
        this.tabla[indice] = valor;
    }

    obtener(clave) {
        const indice = this.hash(clave);
        return this.tabla[indice];
    }
}

// Ejemplo de uso
const tabla = new TablaHash();
tabla.insertar("clave1", "valor1");
tabla.insertar("clave2", "valor2");
console.log(tabla.obtener("clave1")); // Imprime "valor1"

class TablaHash:
    def __init__(self):
        self.tabla = [None] * 10

    def hash(self, clave):
        return sum(ord(c) for c in clave) % 10

    def insertar(self, clave, valor):
        indice = self.hash(clave)
        self.tabla[indice] = valor

    def obtener(self, clave):
        indice = self.hash(clave)
        return self.tabla[indice]

# Ejemplo de uso
tabla = TablaHash()
tabla.insertar("clave1", "valor1")
tabla.insertar("clave2", "valor2")
print(tabla.obtener("clave1")) # Imprime "valor1"
